def find_claim_without_overlap!
  PARSED_FABRIC_CUTS.each.with_index(1) do |(left, top, width, height), index|
    catch :overlap do
      width.times do |w|
        height.times do |h|
          throw :overlap if FABRIC[left + w][top + h] > 1
        end
      end
      return index
    end
  end
end
